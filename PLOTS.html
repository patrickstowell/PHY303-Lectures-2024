#Saxon Wood Potential

def sax(r, A):
    p0 = 1.0
    a = 1.07 * A**(1.0/3.0)
    d = 0.54
    p = p0/(1.0+np.exp((r-a)/d))
    return p
    
x = np.linspace(0.0,10.0,100)

plt.grid()

def sigma(theta):
    # (z*z*Z*Z*alpha*alpha*hbar*hbar*c*c/(4*E*E)) *
    sint = np.sin(theta/2.0)

def volume(A):
    return 15.56 * A

def surface(A):
    return -17.23 * A**(2.0/3.0)

def coulomb(A,Z):
    return -0.72*Z*(Z-1) / A**(1.0/3.0)

def assym(A,Z):
    p_aa = 93.14
    B_a=-p_aa*(Z-A/2)**2/A
    return B_a

def pair(A,Z):
    p_ap=12.0
    B_p=-p_ap/A**(1/2) 
    N=A-Z
    
    CV = np.array(Z%2 + N%2)
    B_p = np.array(B_p)
    print(CV)
    B_p[ CV == 1 ] = -B_p[ CV == 1 ]
    B_p[ CV == 0 ] = 0

    return B_p
